#include<stdio.h>
int main()
{
    int num = 10001;
    int arr[num];
    int i,n;
    scanf("%d",&num);
    if(num==2){
        printf("%d\n",num);
    }
    for(i=0;i<num;i++){
        arr[i]=1;
    }
     
    for(n=2;n<num;n++){
        if(arr[n]){
            for(i=2;i<num;i++){
                if(i*n >10000){//i*n大于你的数组就要结束啊，你一个大于10000的数放在只有a[10000]的哪里
                    break;
                }
                else
                  arr[i*n]=0;
            }
        }
    }
     
    for(i=2;i<num;i++){
        if(arr[i]){
            printf("%d\n",i);
        }
    }
    printf("\n");
    return 0;
}
